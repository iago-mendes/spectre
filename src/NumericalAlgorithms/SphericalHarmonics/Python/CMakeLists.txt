set(LIBRARY "PySphericalHarmonics")

spectre_python_add_module(
  SphericalHarmonics
  LIBRARY_NAME ${LIBRARY}
  SOURCES
  Bindings.cpp
  Spherepack.cpp
  Strahlkorper.cpp
  StrahlkorperFunctions.cpp
  PYTHON_FILES
  __init__.py
)

  ${LIBRARY}
  INCLUDE_DIRECTORY ${CMAKE_SOURCE_DIR}/src
  HEADERS
  Spherepack.hpp
  Strahlkorper.hpp
  StrahlkorperFunctions.hpp
  )

  ${LIBRARY}
  PRIVATE
  DataStructures
  SphericalHarmonics
  SphericalHarmonicsIO
  pybind11::module
)

spectre_python_add_dependencies(
  ${LIBRARY}
  PyDataStructures
  PyTensor
  )
